<script>
const textToScatter = "Valeria Arroyo es GEI";
const numberOfTexts = 100; // Aumentado para llenar más la
pantalla
const container =
document.getElementById('text-scatter-container');
let textElements = []; // Almacenar referencias a los
elementos de texto
// Función para generar un número aleatorio entre min y max
function getRandom(min, max) {
return Math.random() * (max - min) + min;
}
// Genera y posiciona los textos de forma aleatoria con
animación
for (let i = 0; i < numberOfTexts; i++) {
const textDiv = document.createElement('div');
textDiv.textContent = textToScatter;
textDiv.classList.add('scattered-text');
// Define posiciones iniciales y finales para la animación
const startX = getRandom(-20, 120) + 'vw'; // Puede
empezar ligeramente fuera de la pantalla
const startY = getRandom(-20, 120) + 'vh';
const endX = getRandom(-20, 120) + 'vw';
const endY = getRandom(-20, 120) + 'vh';
const startRot = getRandom(-360, 360) + 'deg';
const endRot = getRandom(-360, 360) + 'deg';
// Define la duración y el retraso de la animación para
cada texto
// Más rápido para la fase inicial de 30 segundos
const duration = getRandom(5, 10) + 's';
const delay = getRandom(0, 5) + 's';
// Establece las variables CSS personalizadas para cada
animación
textDiv.style.setProperty('--start-x', startX);
textDiv.style.setProperty('--start-y', startY);
textDiv.style.setProperty('--end-x', endX);
textDiv.style.setProperty('--end-y', endY);
textDiv.style.setProperty('--start-rot', startRot);
textDiv.style.setProperty('--end-rot', endRot);
textDiv.style.animationDuration = duration;
textDiv.style.animationDelay = delay;
container.appendChild(textDiv);
textElements.push(textDiv); // Guarda la referencia
}
/**
* Activa la animación de explosión para todos los textos.
*/
function activateExplosion() {
textElements.forEach(textDiv => {
// Detener la animación actual de flotación
textDiv.style.animation = 'none';
// Forzar un reflow para aplicar el 'animation: none'
antes de la transición
textDiv.offsetHeight; // Esto es una técnica para
forzar que el navegador aplique los cambios de estilo inmediatamente
// Añadir la clase para la transición de explosión
textDiv.classList.add('exploding-text');
// Establecer el estado final de la explosión
const explodeX = getRandom(-200, 300) + 'vw';
const explodeY = getRandom(-200, 300) + 'vh';
const explodeScale = getRandom(2, 5); // Escala para
que se vean más grandes al explotar
const explodeRot = getRandom(-720, 720) + 'deg'; //
Rotación aleatoria al explotar
textDiv.style.transform = `translate(${explodeX},
${explodeY}) scale(${explodeScale}) rotate(${explodeRot})`;
textDiv.style.opacity = '0'; // Se desvanecen
});
}
// Temporizador para activar la explosión después de 30
segundos
const explosionDelay = 30 * 1000; // 30 segundos
setTimeout(activateExplosion, explosionDelay);
// Temporizador para recargar la página después de la
explosión (ej. 1.5 segundos después de la explosión)
// Asegúrate de que este tiempo sea mayor que la duración de
la transición 'exploding-text'
const reloadDelay = explosionDelay + 1500; // 30 segundos +
1.5 segundos de explosión
setTimeout(() => {
location.reload();
}, reloadDelay);
</script>
